Scene description file grammar

file ::=3D camera background materials = group
camera ::=3D orthographicCamera
orthographicCamera ::=3D OrthographicCamera {
    center Vec3f
    direction Vec3f
    up Vec3f
    size float
}
background ::=3D Background {
    color Vec3f
}
materials ::=3D Materials {
    numMaterials int
    Material ^ numMaterials
}
Material ::=3D Material {
    diffuseColor Vec3f
}
group ::=3D Group {
    numObjects int
    (object3D | materialIndex object3D) ^ numObjects
}
object3D ::=3D group | sphere
sphere ::=3D Sphere {
    center Vec3f
    radius float
}
materialIndex ::=3D MaterialIndex int=20
Vec3f ::=3D float float float
